279  inline constexpr float Vector3Ref::Magnitude() const noexcept
280  {
281  float f = x * x + y * y + z * z;
282  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);
283  float bitF = *reinterpret_cast<float*>(&i);
284 
285  bitF *= (1.5f - 0.5f * f * bitF * bitF);
286  return f * bitF * (1.5f - 0.5f * f * bitF * bitF);
287  }
