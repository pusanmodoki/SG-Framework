208  inline void FileWriteStream::Write(const sgstring & write)
209  {
210  //変換

212  int outLength = WideCharToMultiByte(CP_ACP, 0, write.c_str(), inLength, 0, 0, nullptr, nullptr);
213  std::string result(outLength, L'\0');
214  if (outLength) WideCharToMultiByte(CP_UTF8, 0, write.c_str(), inLength, &result[0], outLength, nullptr, nullptr);
215  //書き込み
