116  {
117  ushort oldFlag = 0, newFlag = 0;
118 
119  oldFlag = 0;
120  newFlag = m_cWriteLockFlag;
121 
122  do
123  {
124  //ReadLockの場合Readカウンタを加算した値にする
125  if (IS_FALSE(m_isLockWrite))
126  {
127  oldFlag = m_atomic.load() - (m_atomic.load() & m_cWriteLockFlag);
128  newFlag = oldFlag + 1;
129  }
130  }
131  while (1 ^ m_atomic.compare_exchange_weak(oldFlag, newFlag));
132  }
