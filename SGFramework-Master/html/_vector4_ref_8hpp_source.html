256  inline constexpr float Vector4Ref::Magnitude() const noexcept
257  {
258  float f = x * x + y * y + z * z + w * w;
259  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);
260  float bitF = *reinterpret_cast<float*>(&i);
261 
262  bitF *= (1.5f - 0.5f * f * bitF * bitF);
263  return f * bitF * (1.5f - 0.5f * f * bitF * bitF);
264  }
