129 
130  //拡張子取得
131  sgstring extension = name().my_functions.FindExtension();
132  if (extension == L"")
133  {
134  setIsOpenFailed(true);
135  throw InvalidArgument(L"Error!! TextureAssetX11->Open",
136  L"Invalid extension, extension : " + extension);
137  }
138 
139  //dds
140  if (extension == L".dds")
141  {
142  result = DirectX::CreateDDSTextureFromFile(
143  SGFramework::Administrator::GraphicsX11::m_device,

145 
146  //失敗ならthrow
147  if (FAILED(result))
148  {
149  setIsOpenFailed(true);
150  throw InvalidArgument(L"Error!! TextureAssetX11->Open",
151  L"CreateDDSTextureFromFile failed", result);
152  }
153  }
154  //tga
155  else if (extension == L".tga")
156  {
157  DirectX::TexMetadata metaData;
158  UniquePointer<DirectX::ScratchImage> image(

160 
161  DirectX::GetMetadataFromTGAFile(name().c_str(), metaData);
162  result = DirectX::LoadFromTGAFile(name().c_str(), &metaData, *image);
163 
164  //失敗ならthrow
165  if (FAILED(result))
166  {
167  setIsOpenFailed(true);
168  throw InvalidArgument(L"Error!! TextureAssetX11->Open",
169  L"LoadFromTGAFile failed", result);
170  }
171 
172  result = DirectX::CreateShaderResourceView(SGFramework::Administrator::GraphicsX11::m_device,
173  image->GetImages(), image->GetImageCount(), metaData, &m_shaderResourceView);
174 
175  //失敗ならthrow

177  {
178  setIsOpenFailed(true);
179  throw InvalidArgument(L"Error!! TextureAssetX11->Open",

182  }
