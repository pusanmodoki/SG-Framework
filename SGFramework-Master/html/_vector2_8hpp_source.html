531  inline float Vector2::Angle(const Vector2& from, const Vector2& to) noexcept
532  {
533  float cosSita = Dot(from, to) / (from.Magnitude() * to.Magnitude());
534  float ret = std::acosf(cosSita) * MathAF::cRadToDeg;
535  ret += 90.0f;

545  inline constexpr float Vector2::Distance(const Vector2& from, const Vector2& to) noexcept
546  {

548  + (from.y - to.y) * (from.y - to.y);
549  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);

554  }
