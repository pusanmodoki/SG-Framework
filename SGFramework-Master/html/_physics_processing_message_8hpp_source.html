49  //引数1: message
50  //引数1: message type
51  inline void Physics::ProcessingRigidBodyDynamicMessage(const Detail::Physics::PhysicsMessage & message, uint messageType)
52  {
53  using namespace Detail::Physics;
54 
55  auto dynamic = m_actors.at(message.instanceID).actorDynamic;
56  if (dynamic == nullptr) return;
57 
58  //-------------------------------------------------------------------------------
59  //add force flag
60  if (messageType & MessageType::RigidBodyDynamicAddForceFlag)
61  {
62  //add force
63  if (messageType & MessageType::RigidAddForceFO)
64  {
65  dynamic->addForce(message.attachment.force.force.ToPhysX(),
66  static_cast<physx::PxForceMode::Enum>(message.attachment.force.forceMode),
67  message.attachment.force.isAutoAwake);
68  }
69  //add torque
70  else if (messageType & MessageType::RigidAddTorqueFO)
71  {
72  dynamic->addTorque(message.attachment.force.force.ToPhysX(),
73  static_cast<physx::PxForceMode::Enum>(message.attachment.force.forceMode),
74  message.attachment.force.isAutoAwake);
75  }
76  //clear force
77  else if (messageType & MessageType::RigidClearForce)
78  {
79  dynamic->clearForce();
80  }
81  //clear torque
82  else if (messageType & MessageType::RigidClearTorque)
83  {
84  dynamic->clearTorque();
85  }
86  }
87  //-------------------------------------------------------------------------------
88  //set velocity flag
89  else if (messageType & MessageType::RigidBodyDynamicVelocityFlag)
90  {
91  //set velocity
92  if (messageType & MessageType::RigidSetVelocityVL)
93  {
94  dynamic->setLinearVelocity(message.attachment.velocity.velocity.ToPhysX(),
95  message.attachment.velocity.isAutoAwake);
96  }
97  //set anguler velocity
98  else if (messageType & MessageType::RigidSetAngulerVelocityVL)
99  {
100  dynamic->setAngularVelocity(message.attachment.velocity.velocity.ToPhysX(),
101  message.attachment.velocity.isAutoAwake);
102  }
103  //set velocity max speed
104  else if (messageType & MessageType::RigidSetMaxSpeedFO)
105  {
106  dynamic->setMaxLinearVelocity(message.attachment.floatData);
107  }
108  //set anguler velocity max speed
109  else if (messageType & MessageType::RigidSetAngulerMaxSpeedFO)
110  {
111  dynamic->setMaxAngularVelocity(message.attachment.floatData);
112  }
113  //set velocity damping
114  else if (messageType & MessageType::RigidSetDampingFO)
115  {
116  dynamic->setLinearDamping(message.attachment.floatData);
117  }
118  //set anguler velocity damping
119  else if (messageType & MessageType::RigidSetAngulerDampingFO)
120  {
121  dynamic->setAngularDamping(message.attachment.floatData);
122  }
123  }
124  //-------------------------------------------------------------------------------
125  //other flag
126  else if (messageType & MessageType::RigidBodyDynamicOtherFlag)
127  {
128  //set lock flags (use lock)
129  if (messageType & MessageType::RigidSetLockFlagsLO)
130  {
131  dynamic->setRigidDynamicLockFlag(
132  static_cast<physx::PxRigidDynamicLockFlag::Enum>(message.attachment.lock.flag),
133  message.attachment.lock.flag);
134  }
135  //set lock flags (use byte)
136  else if (messageType & MessageType::RigidSetLockFlagsBY)
137  {
138  dynamic->setRigidDynamicLockFlags(
139  physx::PxRigidDynamicLockFlags(message.attachment.byteData));
140  }
141  //set mass
142  else if (messageType & MessageType::RigidSetMassFL)
143  {
144  dynamic->setMass(message.attachment.floatData);
145  }
146  //set Is Kinematic
147  else if (messageType & MessageType::RigidSetKinematicBO)
148  {
149  //CCDとは併用できない
150  if (IS_TRUE(message.attachment.booleanData))
151  dynamic->setRigidBodyFlag(physx::PxRigidBodyFlag::eENABLE_CCD, false);
152 
153  dynamic->setRigidBodyFlag(physx::PxRigidBodyFlag::eKINEMATIC,
154  message.attachment.booleanData);
155  }
156  //set Is CCD
157  else if (messageType & MessageType::RigidSetCcdBO)
158  {
159  //Kinematicとは併用できない
160  if (IS_TRUE(message.attachment.booleanData))
161  dynamic->setRigidBodyFlag(physx::PxRigidBodyFlag::eKINEMATIC, false);
162 
163  dynamic->setRigidBodyFlag(physx::PxRigidBodyFlag::eENABLE_CCD,
164  message.attachment.booleanData);
165  }
166  //set Is Sleep
167  else if (messageType & MessageType::RigidSetSleepBO)
168  {
169  //kinematicには無効
170  if (IS_TRUE(dynamic->getRigidBodyFlags().isSet(physx::PxRigidBodyFlag::eKINEMATIC)))
171  return;
172 
173  //now sleeping
174  bool isNowResult = dynamic->isSleeping();
175  //true->false = sleep
176  if ((message.attachment.booleanData ^ isNowResult) & isNowResult)
177  dynamic->putToSleep();
178  //false->true = wakeUp
179  else if (((message.attachment.booleanData ^ isNowResult) & message.attachment.booleanData))
180  dynamic->wakeUp();
181  }
182  //wake up
183  else if (messageType & MessageType::RigidWakeUp)
184  {
185  dynamic->wakeUp();
186  }
187  }
188  }

193  //メッセージを処理する (Pointer)
194  //引数1: message
195  //引数1: message type
196  inline void Physics::ProcessingPointerMessage(const Detail::Physics::PhysicsMessage & message, uint messageType)
197  {
198  using namespace Detail::Physics;
199 
200  //register Rigid Body
201  if (messageType & MessageType::RegisterRigidBodyRG)
202  {
203  auto pointer = message.attachment.rigidBody;
204  if (IS_TRUE(pointer.getIsRevocatoin())) return;
205 
206  //Actorを追加
207  pointer->BuildRigidBody();
208  if (pointer->getBaseActor() == nullptr)
209  return;
210 
211  m_scene->addActor(*pointer->getBaseActor());
212 
213  m_actors.try_emplace(message.instanceID,
214  pointer->transform, pointer, pointer->getBaseActor(),
215  pointer->getIsStatic() ? nullptr : reinterpret_cast<physx::PxRigidDynamic*>(pointer->getBaseActor()),
216  IS_FALSE(pointer->getIsStatic()));
217 
218  if (IS_FALSE(pointer->getIsStatic()))
219  m_dynamicActors.try_emplace(message.instanceID, m_actors.at(message.instanceID).rigidBody);
220  }
221  //register Collider
222  else if (messageType & MessageType::RegisterColliderCL)
223  {
224  auto pointer = message.attachment.collider;
225  if (IS_TRUE(pointer.getIsRevocatoin())) return;
226 
227  //Colliderを追加
228  pointer->GetActor();
229  pointer->CreateShape();
230  if (pointer->m_shape == nullptr)
231  return;
232 
233  m_colliders.try_emplace(message.instanceID, pointer, pointer->m_shape);
234  }
235  //unregister Rigid Body
236  else if (messageType & MessageType::UnregisterRigidBody)
237  {
238  if (m_actors.find(message.instanceID) == m_actors.end())
239  return;
240 
241  auto& at = m_actors.at(message.instanceID);
242 
243  //release後マップから削除
244  at.actor->release();
245  if (IS_TRUE(at.isDynamic)) m_dynamicActors.erase(message.instanceID);
246  m_actors.erase(message.instanceID);
247  }
248  //unregister Collider
249  else if (messageType & MessageType::UnregisterCollider)
250  {
251  if (m_colliders.find(message.instanceID) == m_colliders.end())
252  return;
253 
254  auto& at = m_colliders.at(message.instanceID);
255 
256  //release後マップから削除
257  if (at.shape->getActor())
258  at.shape->getActor()->detachShape(*at.shape);
259 
260  at.shape->release();
261  m_colliders.erase(message.instanceID);
262  }
263  }
