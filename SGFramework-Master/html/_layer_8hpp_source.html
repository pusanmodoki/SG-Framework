486  {

488  uint layerID = 1;
489 
490  //Default登録
491  m_layersKeyName.try_emplace(cDefaultLayerName, Layer(cDefaultLayerID, cDefaultLayerName));
492  m_layersKeyID.try_emplace(cDefaultLayerID, Layer(cDefaultLayerID, cDefaultLayerName));
493 
494  //登録ループ
495  for (auto& e : layers.packs)
496  {
497  //Layer登録
498  m_layersKeyName.try_emplace(e.header, Layer(layerID, e.header));
499  m_layersKeyID.try_emplace(layerID, Layer(layerID, e.header));
500  layerID++;
501  }
502 
503  //Table作成ループ
504 
505  //DefaultLayer
506  {
507  uint hitLayerMask = 1;
508  for (uint i = 0, size = static_cast<uint>(layers.packs.size()); i < size; ++i)
509  {
510  if (layers.packs[i].elements.size() >= 1)
511  hitLayerMask |= static_cast<uint>(layers.packs[i].elements[0].valueInt) << (i + 1);
512  }
513  m_layerHitTable.try_emplace(1 << cDefaultLayerID, hitLayerMask);
514  }
515 
516  //OtherLayer
517  for (uint i = 0, size = static_cast<uint>(layers.packs.size()); i < size; ++i)
518  {
519  uint hitLayerMask = 0;
520  //横ループ
521  for (uint k = 0, size1 = static_cast<uint>(layers.packs[i].elements.size()); k < size1; ++k)
522  hitLayerMask |= static_cast<uint>(layers.packs[i].elements[k].valueInt) << k;
523  //縦ループ
524  for (uint k = i + 1; k < size; ++k)
525  {
526  if (layers.packs[k].elements.size() > i + 1)
527  hitLayerMask |= static_cast<uint>(layers.packs[k].elements[i + 1].valueInt) << (i + 1);
528  }
529 
530  m_layerHitTable.try_emplace(1 << (i + 1), hitLayerMask);
531  }
532  }
