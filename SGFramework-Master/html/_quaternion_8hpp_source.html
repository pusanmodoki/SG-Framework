412  {
413  //temp
414  DirectX::XMVECTOR temp1, temp2;
415  temp1 = DirectX::XMLoadFloat3(&fromDirection);
416  temp2 = DirectX::XMLoadFloat3(&toDirection);
417  //acos
418  float dotToAcos = std::acos(DirectX::XMVectorGetX(DirectX::XMVector3Dot(temp1, temp2)));
419 
420  //Set
421  if (dotToAcos > MathAF::cEpsilon)
422  return std::move(DirectX::XMQuaternionRotationAxis(
423  DirectX::XMVector3Normalize(DirectX::XMVector3Cross(temp1, temp2)), dotToAcos));
424  //Identity
