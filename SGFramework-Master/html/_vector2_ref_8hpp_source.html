267  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);
268  float bitF = *reinterpret_cast<float*>(&i);
269 
270  bitF *= (1.5f - 0.5f * f * bitF * bitF);
271  return f * bitF * (1.5f - 0.5f * f * bitF * bitF);
272  }
