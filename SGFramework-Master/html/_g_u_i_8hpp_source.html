302  inline GUI::ConsoleMessage::ConsoleMessage(const std::string& header, const sgstring& message) : message()
303  {
304  //Headerありメッセージ->Color & Headerをmessageと合成しconsole message作成
305  if (sgstring::functions::IsContains(message, m_cLogHerderIdentification))
306  {
307  //Error
308  if (message[0] == m_cLogHerderError)
309  {
310  this->message = m_cLogErrorString + header + message.to_std_string().substr(m_cLogIdentificationSize - 1);
311  color = m_cErrorColor;
312  isUsingColor = true;
313  }
314  //Worning
315  else if (message[0] == m_cLogHerderWorning)
316  {
317  this->message = m_cLogWorningString + header + message.to_std_string().substr(m_cLogIdentificationSize - 1);
318  color = m_cWorningColor;
319  isUsingColor = true;
320  }
321  //Info
322  else if (message[0] == m_cLogHerderInfo)
323  {
324  this->message = m_cLogInfoString + header + message.to_std_string().substr(m_cLogIdentificationSize - 1);
325  color = m_cInfoColor;
326  isUsingColor = true;
327  }
328  //Framework
329  else if (message[0] == m_cLogHerderFramework)
330  {
331  this->message = m_cLogFrameworkString + header + message.to_std_string().substr(m_cLogIdentificationSize - 1);
332  color = m_cFrameworkMessageColor;
333  isUsingColor = true;
334  }
335  }
336  //Headerなしメッセージ->messageのみでconsole message作成
337  else
338  {
339  this->message = header + message.to_std_string();
340  isUsingColor = false;
341  }
342  }
