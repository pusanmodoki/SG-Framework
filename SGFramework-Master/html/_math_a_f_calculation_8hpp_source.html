195  //return: fの平方根, Fast inverse square rootを用いるので精度は低め
196  //引数1: f (radian)
197  inline constexpr float Sqrt(float f) noexcept
198  {
199  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);
200  float bitF = *reinterpret_cast<float*>(&i);
201 
202  bitF *= (1.5f - 0.5f * f * bitF * bitF);
203  return f * bitF * (1.5f - 0.5f * f * bitF * bitF);
204  }

270  inline constexpr int GreatestCommonDenominator(int a, int b) noexcept
271  {
272  // - の値の場合挙動が怪しいので終了
273  if (a < 0 || b < 0) return 0;
274  //bがaより大きい場合逆を実行
275  else if (a < b) return GreatestCommonDenominator(b, a);
276  else
277  {
278  //ユークリッドの互除法
279  while (b != 0)
280  {
281  int temp = a % b;
282  a = b;
283  b = temp;
284  }
285  return a;
286  }
287  }
