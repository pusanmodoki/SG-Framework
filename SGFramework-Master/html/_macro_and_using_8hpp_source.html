315  inline float GetRefreshRate(HWND hWnd)
316  {
317  //設定リフレッシュレート
318  constexpr std::array<int, 5> cRefreshs = { 30, 60, 120, 144, 244 };
319 
320  //リフレッシュレートを取得
321  int result;
322  {
323  HDC hdc = GetDC(hWnd);
324  result = GetDeviceCaps(hdc, VREFRESH);
325  ReleaseDC(hWnd, hdc);
326  }
327 
328  //設定リフレッシュレートと合致するか確認
329  bool isEqual = false;
330  for (const auto& e : cRefreshs)
331  if (result == e) isEqual = true;
332  //一緒ならそのまま返す
333  if (IS_TRUE(isEqual)) return static_cast<float>(result);
334 
335  //一番近い値を探す用
336  int fasts[2] = { 100, -1 };
337  //探索ループ
338  for (int i = 0; i < 5; i++)
339  {
340  int distance = cRefreshs[i] - result;
341  if (distance < 0) distance *= - 1;
342  if (fasts[0] > distance)
343  {
344  fasts[0] = distance;
345  fasts[1] = i;
346  }
347  }
348 
349  //一番近い値を返却
350  return static_cast<float>(cRefreshs[fasts[1]]);
351  }
352 #else
353 #error "GetRefreshRate->Unimplemented"
354  throw std::exception("GetRefreshRate->Unimplemented");
355 #endif
356  }
