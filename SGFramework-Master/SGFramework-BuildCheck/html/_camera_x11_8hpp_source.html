242  //graphics set camera matrix(graphics thread only property)
243  SGF_FUNCTION_PROPERTY void CameraX11::setCameraMatrix()
244  {
245  using GraphicsX11 = SGFramework::Administrator::GraphicsX11;
246 
247  //すでに設定済みならば終了
248  if (m_setCamera == this) return;
249 
250  //設定
251  m_setCamera = this;
252  m_setLayerMask = m_buffersUseGraphics.layerMask;
253 
254  //Create Projection
255  if (IS_FALSE(m_isCreateProjection.load()))
256  {
257  m_projectionMatrix = Matrix::ProjectionMatrix(m_buffersUseGraphics.fov,
258  m_buffersUseGraphics.viewPort.drawSizePerScreen.x / m_buffersUseGraphics.viewPort.drawSizePerScreen.y,
259  m_buffersUseGraphics.nearClip, m_buffersUseGraphics.farClip);
260 
261  AtomicOperation::Init(m_isCreateProjection, true);
262  }
263 
264  //Create View
265  if (IS_FALSE(m_isCreateView.load()))
266  {
267  //LookAt
268  if (IS_TRUE(m_isLookAtUseGrapics))
269  m_viewMatrix = Matrix::ViewMatrix(transform->m_drawMatrix.getTranslate(),
270  m_lookAt, transform->m_drawMatrix.getUp());
271  //Auto Focus
272  else
273  m_viewMatrix = Matrix::ViewMatrixForDirection(transform->m_drawMatrix.getTranslate(),
274  transform->m_drawMatrix.getForward(),
275  transform->m_drawMatrix.getUp());
276 
277  AtomicOperation::Init(m_isCreateView, true);
278  }
