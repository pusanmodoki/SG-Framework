53  //BGMストリーミングなどの随時割り振り型別スレッド関数を適切なスレッドに割り振る
54  inline static uint64 AutoThreadAllocation(const HANDLE& threadHandle)
55  {
56  static std::atomic<uint> allocation = 0; //シフト用
57 
58  //nullなら終了
59  if (threadHandle == nullptr) return 0;
60 
61  //マスク取得
62  uint64 result = 1Ui64 << AtomicOperation::Add(allocation, 1) %
63  MathAF::Min(std::thread::hardware_concurrency(), m_numMaxUsingThreads);
64 
65  //アフィニティ設定
66  SetThreadAffinityMask(threadHandle, result);
67 
68  return result;
69  }
