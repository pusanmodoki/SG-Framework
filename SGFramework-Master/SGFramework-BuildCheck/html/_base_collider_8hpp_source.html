98 
99  //find?

101  bool isFindStatic = rigidStatic.getIsValid();
102 
103  //find dynamic
104  if ((isFindDynamic ^ isFindStatic) & isFindDynamic)
105  {
106  m_rigidBodyTransform = rigidDynamic->transform;
107  m_rigidBody = rigidDynamic.DownCast<BaseClass::BaseRigidBody>();
108  m_actor = &rigidDynamic->getPhysX();
109  }
110  //find static
111  else if ((isFindDynamic ^ isFindStatic) & isFindStatic)
112  {
113  m_rigidBodyTransform = rigidStatic->transform;
114  m_rigidBody = rigidStatic.DownCast<BaseClass::BaseRigidBody>();
115  m_actor = &rigidStatic->getPhysX();
116  }
117  //Both Find->どちらが直近か判定
118  else if (isFindDynamic & isFindStatic)

121  for (auto parent = transform;
122  IS_TRUE(parent.getIsValid()); parent = parent->getParent())
123  {
124  //どちらか見つかれば終了
125  if (rigidStatic->transform->instanceID() == parent->instanceID()
126  || rigidDynamic->transform->instanceID() == parent->instanceID())
127  {
128  m_rigidBodyTransform = parent;
129  m_actor = &rigidStatic->getPhysX();

253  void UpdatePhysics()
254  {
255  //un linked
256  if ((m_isLinkRigidBody & m_rigidBodyTransform.getIsValid()) ^ true)
257  {
258  //not simulation
259  if (m_shape) m_shape->setFlags(static_cast<physx::PxShapeFlags>(0));
260  m_isUnLinked = true;
261  m_isLinkRigidBody = false;
262  return;
263  }
264 
265  //un linked->linked
266  if (IS_TRUE(m_isUnLinked))
267  {
268  //set flags
269  m_shape->setFlag(physx::PxShapeFlag::eSIMULATION_SHAPE, !m_isTrigger);
270  m_shape->setFlag(physx::PxShapeFlag::eSCENE_QUERY_SHAPE, true);
271  m_shape->setFlag(physx::PxShapeFlag::eTRIGGER_SHAPE, m_isTrigger);
272  m_shape->setFlag(physx::PxShapeFlag::eVISUALIZATION, IsDebug::isValue());
273  //linked
274  m_isUnLinked = false;
275  }
276  //not shared & not equal attach
277  else if (m_shape->isExclusive() & (m_isEqualRigidTransform ^ 1))
278  {
279  //not plane
280  if (m_type != ColliderType::Plane)
281  {
282  //Change scale
283  ChangeScale();
284 
285  //equal rigidbody transfoem
286  if (IS_TRUE(m_isEqualRigidTransform))
287  m_shape->setLocalPose(physx::PxTransform(CalculateCenter().ToPhysX(), Const::Quaternion::identity.ToPhysX()));
288  //not equal rigidbody transfoem
289  else
290  m_shape->setLocalPose(physx::PxTransform(
291  (transform->getWorldPosition() - transform->getParent()->getWorldPosition() + CalculateCenter()).ToPhysX(),
292  (transform->getWorldRotation() * transform->getParent()->getWorldRotation().getInverse()).ToPhysX()));
293  }
294  //plane
295  else
296  {
297  m_shape->setLocalPose(physx::PxTransformFromPlaneEquation(
298  physx::PxPlane(
299  (transform->getWorldPosition() + CalculateCenter()).ToPhysX(),
300  (transform->getWorldRotation() * Const::Vector3::down).ToPhysX())));
301  }
302  }
