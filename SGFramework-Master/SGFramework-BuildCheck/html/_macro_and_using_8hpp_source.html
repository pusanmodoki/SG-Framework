300  //[GetRefreshRate]
301  //return: 30, 60, 120, 144, 244の中からGetDeviceCapsで取得したリフレッシュレートに最も近い値
302  //argument 1: GetDeviceCapsで使用するHWND
303  inline float GetRefreshRate(HWND hWnd)
304  {
305  //設定リフレッシュレート
306  constexpr std::array<int, 5> cRefreshs = { 30, 60, 120, 144, 244 };
307 
308  int result;
309  bool isEqual = false;
310 
311  //Get ...Windows Version
312 #if defined(SGF_PLATFORM_WINDOWS)
313  HDC hdc = GetDC(hWnd);
314  result = GetDeviceCaps(hdc, VREFRESH);
315  ReleaseDC(hWnd, hdc);
316 #else
317 #error "GetRefreshRate->Unimplemented"
318  throw std::exception("GetRefreshRate->Unimplemented");
319 #endif
320  //設定リフレッシュレートと合致するか確認
321  for (const auto& e : cRefreshs)
322  if (result == e) isEqual = true;
323  //一緒ならそのまま返す
324  if (IS_TRUE(isEqual)) return static_cast<float>(result);
325 
326  //一番近い値を探す用
327  int fasts[2] = { 100, -1 };
328  //探索ループ
329  for (int i = 0; i < 5; i++)
330  {
331  int distance = cRefreshs[i] - result;
332  if (distance < 0) distance *= - 1;
333  if (fasts[0] > distance)
334  {
335  fasts[0] = distance;
336  fasts[1] = i;
337  }
338  }
339 
340  //一番近い値返却
341  return static_cast<float>(cRefreshs[fasts[1]]);
342  }
343  }
