247  inline void Time::StartTimeProcessing(const ReadElement::Pack& isNoRestrictionsFPS,
248  const ReadElement::Pack& fixedInterval, const ReadElement::Groop& layers, bool isRemote)
249  {
250  //リフレッシュレート取得
251  if (IS_FALSE(isRemote))
252  {
253  m_refreshRate = Administrator::GetRefreshRate(Window::HwndHolder::main);
254  }
255  else
256  {
257  m_refreshRate = 30.0f;
258  }
259 
260  if (isNoRestrictionsFPS[0].valueInt)
261  m_refreshRate = VariableLimit::cFloatMax;
262  //Interval代入
263  m_fixedInterval = fixedInterval[0].valueFloat;
264 
265  //RootLayer設定
266  m_rootLayer = SharedPointer<TimeLayer>(new TimeLayer());
267  m_rootLayer->InitRoot();
268  m_layerMap.try_emplace(0, m_rootLayer);
269  m_layerCount = 1;
270 
271  for (auto& e : layers.packs)
272  {
273  auto pointer = SharedPointer<TimeLayer>(new TimeLayer());
274  auto parent = std::move(e.elements.size() >= 2
275  ? FindTimeLayer(e[1].valueString) : m_rootLayer);
276 
277  pointer->Init(pointer, parent, e.header, e[0].valueFloat, m_layerCount);
278  m_layerMap.try_emplace(m_layerCount++, pointer);
279  }
280 
281  m_fpsLimiter.Start(m_refreshRate); //FPS制限開始
282  m_fpsInstrumentationLimiter.Start(1.0f); //FPS計測制限開始
283 
284  m_startUpTime = m_timeGetter(); //Start実行時刻
285  //代入
286  m_frequencyTime = m_startUpTime;
287  m_updateStartTime = m_startUpTime;
288  m_fixedStartTime = m_startUpTime;
289 
290  while (m_time < 1.0f)
291  {
292  if (IS_TRUE(Time::FixedUpdate()))
293  Time::FixedLastUpdate();
294  Time::Update();
295  }
296  }

299  //[FixedUpdate]
300  //Fixedタイム処理の更新を行う
301  //return: Fixed更新フレームか否か, n回trueを繰り返す
302  inline bool Time::FixedUpdate()
303  {
304  static double delta = 0; //Intervl確認ループ用
305 
306  //Get Time
307  m_frequencyTime = m_timeGetter();
308 
309  //FPS Update
310  if (m_fpsInstrumentationLimiter(m_frequencyTime))
311  {
312  //Set FPS
313  m_fps = static_cast<float>(static_cast<double>(m_frameCount) / (m_frequencyTime - m_fpsInstrumentationLimiter.previousStartTime));
314  //Reset Counter
315  m_frameCount = 0;
316  if (m_fps < m_refreshRate)
317  m_fpsLimiter.ChangeFPS(m_refreshRate * 1.01f);
318  else if (m_fps > m_refreshRate + 1.0f)
319  m_fpsLimiter.ChangeFPS(m_refreshRate* 0.99f);
320  }
321 
322  if (m_frequencyTime - m_fixedStartTime > m_fixedInterval * 3)
323  {
324  m_fpsLimiter.Start(m_refreshRate); //FPS制限開始
325  m_fpsInstrumentationLimiter.Start(1.0f); //FPS計測制限開始
326  //fixed更新強制
327  m_fixedStartTime = m_frequencyTime - m_fixedInterval;
328  m_fixedCount = 0;
329  }
330 
331  //FixedCountがまだ残っている
332  if (m_fixedCount > 1)
333  {
334  --m_fixedCount; //カウンター減算
335  //Fixed更新回数がまだあり累計更新時間がFixedIntervalを超過した場合更新を諦める
336  if (m_fixedCount > 1 && m_timeGetter() - m_fixedStartTime > m_fixedInterval)
337  m_fixedCount = 0;
338  //true返却
339  return m_isFixedUpdateFreame = true;
340  }
341 
342  //現在のFixed差分時間取得
343  delta = m_frequencyTime - m_fixedStartTime;
344  //カウンター初期化
345  m_fixedCount = 0;
346 
347  //Interval確認ループ
348  while (delta > m_fixedInterval)
349  {
350  ++m_fixedCount; //カウンター加算
351  delta -= m_fixedInterval; //delta減算
352  }
353  //カウンター0超過 -> Fixed Update Frame
354  if (m_fixedCount > 0)
355  {
356  //Get FixedApplicationElapasedTime
357  m_fixedTime = static_cast<float>(m_frequencyTime - m_startUpTime);
358  //Get FixedDeltaTime
359  m_unScaledFixedDeltaTime = static_cast<float>(m_frequencyTime - m_fixedStartTime);
360  //delta設定
361  m_fixedDeltaTime = m_unScaledFixedDeltaTime * m_worldTimeScale;
362 
363  m_time = m_fixedTime; //Time代入
364  m_fixedStartTime += m_fixedInterval; //StartTime更新
365  return m_isFixedUpdateFreame = true; //true返却
366  }
367 
368  //Updateは許す
369  m_isAllowUpdate = true;
370  //false返却
371  return m_isFixedUpdateFreame = false;
372  }
373  //----------------------------------------------------------------------------------
374  //[FixedLastUpdate]
375  //Fixedタイム処理の最終更新を行う
376  //補正モードの確認を行うためフレーム実行処理の中に入れること
377  inline void Time::FixedLastUpdate()
378  {
379  //FPS Update
380  if (m_fpsInstrumentationLimiter(m_frequencyTime))
381  {
382  //Set FPS
383  m_fps = static_cast<float>(static_cast<double>(m_frameCount) / (m_frequencyTime - m_fpsInstrumentationLimiter.previousStartTime));
384  //Reset Counter
385  m_frameCount = 0;
386  if (m_fps < m_refreshRate)
387  m_fpsLimiter.ChangeFPS(m_refreshRate * 1.01f);
388  else if (m_fps > m_refreshRate + 1.0f)
389  m_fpsLimiter.ChangeFPS(m_refreshRate* 0.99f);
390  }
391 
392  //FixedUpdate実行秒数がInterval超過->Update更新を許可しない
393  //超過していなければUpdate更新を許可する
394  if (m_timeGetter() - m_fixedStartTime < m_fixedInterval)
395  m_isAllowUpdate = true;
396  else
397  m_isAllowUpdate = false;
398  }

401  //[Update]
402  //タイム処理の更新を行う
403  //return: 更新フレームか否か
404  inline bool Time::Update()
405  {
406  //Update更新が許可されていない場合はfalse返却
407  if (m_isAllowUpdate ^ true) return m_isUpdateFreame = false;
408 
409  //Get ApplicationElapasedTime
410  m_time = static_cast<float>(m_frequencyTime - m_startUpTime);
411 
412  //UpdateFrame
413  if (m_fpsLimiter(m_frequencyTime))
414  {
415  //Get
416  m_worldUnScaledDeltaTime = static_cast<float>(m_frequencyTime - m_updateStartTime);
417  if (m_worldUnScaledDeltaTime > 1.0f)
418  m_worldUnScaledDeltaTime = static_cast<float>(m_fpsLimiter.interval());
419 
420  m_worldDeltaTime = m_worldUnScaledDeltaTime * m_worldTimeScale;
421  //TimeLayer Update
422  for (auto& it : m_layerMap) it.second->Update();
423 
424  //Save Time
425  m_updateStartTime = m_frequencyTime;
426  //FPS Counter++
427  m_frameCount++;
428 
429  //return true
430  return m_isUpdateFreame = true;
431  }
432  //NotUpdateFrame
433  else
434  {
435  //return false
436  return m_isUpdateFreame = false;
437  }
438  }
