252  inline void GabageCollector::GabageCollection()
253  {
254  //Lock
255  AtomicOperation::LockGuard guard(m_lock);
256 
257  //Start Gabage coollection
258  m_isNowGabageCollection = true;
259  //num deleteing objects
260  uint deleteCounter = 0;
261  //num delete max
262  uint numDeleteMax = m_numDeleteOfOneFlame + ((m_numDeleteCapacity < deleteCounter + m_numDeleteObjects) ? deleteCounter + m_numDeleteObjects : 0);
263  size_t allSize = 0;
264 
265  do
266  {
267  DeletePointers(deleteCounter, numDeleteMax);
268 
269  allSize = m_deleteAutoUniques.size() + m_deleteAutoShareds.size()
270  + m_deleteAutoCloneShareds.size() + m_deleteWeaks.size();
271 
272  if (IS_TRUE(m_isManualCollectionUsedNextFrame))
273  allSize += m_deleteManualUniques.size() + m_deleteManualShareds.size()
274  + m_deleteManualCloneShareds.size();
275  } while (numDeleteMax > deleteCounter && allSize > 0);
276 
277  //Clone Update
278  auto cloneReferenceEnd = m_cloneSharedReferences.end();
279  for (auto& e : m_cloneShareds)
280  {
281  //Clone Unique & Alive Pointer -> Update
282  if ((e.second.cloneMode & (m_cCloneModeUnique | m_cCloneModeShared))
283  && m_cloneSharedReferences.find(e.second.cloneSourcePointer) != cloneReferenceEnd)
284  memcpy_s(e.first, e.second.copySize, e.second.cloneSourcePointer, e.second.copySize);
285  }
286 
287  //Init Delete flags
288  m_deleteFlags = (IS_TRUE(m_isManualCollectionUsedNextFrame) ? (m_deleteFlags & m_cExcludeNotManual) : 0);
289  //Init Manual flags
290  m_isManualCollectionUsedNextFrame = false;
291  //End gabage collection
292  m_isNowGabageCollection = false;
293  }
