58  //引数3: 終了後停止するか否か, default = false
59  inline void ChangeSpeedSmoothly(float target, float seconds, bool isStop = false)
60  {
61  using namespace Detail::Audio;
62 
63  //現在の値
64  float speed = getPlayBackSpeed();
65  //同じなら終了
66  if (target == speed) return;
67 
68  //Message用
69  AudioMessage message(MessageType::SmoothFlag | MessageType::AddSmoothAudio);
70 
71  //Messageの添付情報を代入
72  message.attachment.smoothAudio.asset = m_thisPointer.DynamicCast<BaseAudioAsset>();
73  message.attachment.smoothAudio.smoothType = SmoothType::Speed;
74  message.attachment.smoothAudio.now = speed;
75  message.attachment.smoothAudio.target = target;
76  message.attachment.smoothAudio.speed = (target - speed) / seconds;
77  message.attachment.smoothAudio.isStop = isStop;
78 
79  //Message送信
80  Audio::SendAudioMessage(message);
81  }
