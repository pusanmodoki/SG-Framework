405  inline static void SendPhysicsMessage(const Detail::Physics::PhysicsMessage& message)
406  {
407  auto type = message.type;
408 
409  AtomicOperation::LockAtomic(m_lockMessages);
410 
411  if (type & Detail::Physics::MessageType::PointerMessageFlag)
412  {
413  type ^= Detail::Physics::MessageType::PointerMessageFlag;
414 
415  if (type & Detail::Physics::MessageType::UnregisterCollider
416  || type & Detail::Physics::MessageType::UnregisterRigidBody)
417  {
418  if (m_preferentRegisterInstanceIDs.find(message.instanceID)
419  != m_preferentRegisterInstanceIDs.end())
420  {
421  m_preferentRegisterInstanceIDs.erase(message.instanceID);
422 
423  for (size_t i = 0, max = m_preferentMessages.size(); i < max; ++i)
424  {
425  if (m_preferentMessages[i].instanceID == message.instanceID)
426  {
427  m_preferentMessages[i].Release();
428  m_preferentMessages.erase(m_preferentMessages.begin() + i);
429  break;
430  }
431  }
432  }
433  else
434  {
435  m_preferentMessages.emplace_back(message);
436  m_preferentRegisterInstanceIDs.emplace(message.instanceID);
437  }
438  }
439  else if (type & Detail::Physics::MessageType::RegisterColliderCL
440  || type & Detail::Physics::MessageType::RegisterRigidBodyRG)
441  {
442  m_preferentMessages.emplace_back(message);
443  m_preferentRegisterInstanceIDs.emplace(message.instanceID);
444  }
445  }
446  else
447  m_messages.emplace_back(message);
448 
449  AtomicOperation::UnlockAtomic(m_lockMessages);
450  }

535  //pair found callback
536  inline physx::PxFilterFlags Physics::SimulationCallback::pairFound(physx::PxU32 pairID,
537  physx::PxFilterObjectAttributes attributes0, physx::PxFilterData filterData0,
538  const physx::PxActor * a0, const physx::PxShape * s0,
539  physx::PxFilterObjectAttributes attributes1, physx::PxFilterData filterData1,
540  const physx::PxActor * a1, const physx::PxShape * s1, physx::PxPairFlags & pairFlags)
541  {
542  //used callback & not equal transforms->register found pair
543  if ((filterData0.word2 | filterData1.word2) && filterData0.word3 != filterData1.word3)
544  {
545  m_foundPairs.emplace_back(HitPair::CalculateHashSeed(filterData0.word3, filterData1.word3),
546  s0->userData, s1->userData, filterData0.word3, filterData1.word3,
547  s0->getFlags().isSet(physx::PxShapeFlag::eTRIGGER_SHAPE),
548  s1->getFlags().isSet(physx::PxShapeFlag::eTRIGGER_SHAPE),
549  filterData0.word2 != 0, filterData1.word2 != 0);
550  return physx::PxFilterFlag::eNOTIFY;
551  }
552  else
553  return physx::PxFilterFlag::eSUPPRESS;
554  }
555  //pair lost callback
556  inline void Physics::SimulationCallback::pairLost(physx::PxU32 pairID,
557  physx::PxFilterObjectAttributes attributes0, physx::PxFilterData filterData0,
558  physx::PxFilterObjectAttributes attributes1, physx::PxFilterData filterData1, bool objectRemoved)
559  {
560  //ハッシュシード生成
561  uint64 id = HitPair::CalculateHashSeed(filterData0.word3, filterData1.word3);
562  //Found
563  auto find = m_hitPairs.find(id);
564 
565  //Pair発見->Lost追加
566  if (find != m_hitPairs.end())
567  m_lostPairs.emplace_back(find->second, objectRemoved, true);
568  else
569  {
570  //FoundPairからも検索, なければ追加
571  for(auto& e : m_foundPairs)
572  if (e.hashID == id)
573  {
574  m_lostPairs.emplace_back(e, objectRemoved, false);
575  return;
576  }
577  }

608  //on trigger event
609  inline void Physics::SimulationEventCallback::onTrigger(physx::PxTriggerPair *pairs, physx::PxU32 count)
610  {
611  //Trigger側FilterData
612  auto triggerFilterData = pairs->triggerShape->getSimulationFilterData();
613  //相手のFilterData
614  auto otherFilterData = pairs->otherShape->getSimulationFilterData();
615 
616  //TriggerEnter & レイヤー判定 & !同一オブジェクト判定->Found登録
617  if ((pairs->status & physx::PxPairFlag::eNOTIFY_TOUCH_FOUND) &&
618  (triggerFilterData.word2 | otherFilterData.word2) &&
619  triggerFilterData.word3 != otherFilterData.word3)
620  {
621  m_foundPairs.emplace_back(HitPair::CalculateHashSeed(triggerFilterData.word3, otherFilterData.word3),
622  pairs->triggerShape->userData, pairs->otherShape->userData, triggerFilterData.word3, otherFilterData.word3,
623  pairs->triggerShape->getFlags().isSet(physx::PxShapeFlag::eTRIGGER_SHAPE),
624  pairs->otherShape->getFlags().isSet(physx::PxShapeFlag::eTRIGGER_SHAPE),
625  triggerFilterData.word2 != 0, otherFilterData.word2 != 0);
626  }
627  //TriggerExit
628  else if ((pairs->status & physx::PxPairFlag::eNOTIFY_TOUCH_LOST))
629  {
630  //ハッシュシード生成
631  uint64 id = HitPair::CalculateHashSeed(pairs->triggerShape->getSimulationFilterData().word3,
632  pairs->otherShape->getSimulationFilterData().word3);
633 
634  //Found
635  auto find = m_hitPairs.find(id);
636  //オブジェクト削除によるExitか判定
637  bool isObjectRemoved = (pairs->flags &
638  (physx::PxTriggerPairFlag::eREMOVED_SHAPE_OTHER |
639  physx::PxTriggerPairFlag::eREMOVED_SHAPE_TRIGGER)) ? true : false;
640 
641  //Pair存在ならLost追加
642  if (find != m_hitPairs.end())
643  m_lostPairs.emplace_back(find->second, isObjectRemoved, true);
644  else
645  {
646  //Foundからも検索->見つければLost追加
647  for (auto& e : m_foundPairs)
648  if (e.hashID == id)
649  {
650  m_lostPairs.emplace_back(e, isObjectRemoved, false);
651  return;
652  }
653  }
654  }
655  }
