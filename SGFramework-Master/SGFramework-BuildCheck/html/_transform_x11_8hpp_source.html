105  {
106  std::thread::id thisID = std::this_thread::get_id();
107  Lock newLock, oldLock;
108  bool result = false;
109 
110  //同じスレッドがロックしていなければロックするまで待機
111  newLock.id = thisID;
112  do
113  {
114  oldLock = m_lock.load();
115  oldLock.id = m_cDefauldID;
116  newLock.counter = oldLock.counter + 1;
117 
118  result = m_lock.compare_exchange_weak(oldLock, newLock);
119  if (IS_FALSE(result))
120  {
121  oldLock = m_lock.load();
122  oldLock.id = thisID;
123  newLock.counter = oldLock.counter + 1;
124  result = m_lock.compare_exchange_weak(oldLock, newLock);
125  }
126  } while (IS_FALSE(result));
127  }
