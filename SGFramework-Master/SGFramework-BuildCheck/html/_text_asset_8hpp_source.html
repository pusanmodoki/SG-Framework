114  setIsOpenFailed(false);

116  //mutex guard

119  //ないかもしらんからとりあえず開いとく
120  File::FileWriteStream fws;
121  fws.Open(name(), false, true);
122  fws.Close();
123 
124  bool isOpened = false;
125  //sec
126  if (name().my_functions.IsEndWith(m_cIdentificationSecOrSecl[0]))
127  {
128  isOpened = true;
129  File::SecFileRW::ReadNotLocked(name(), m_textArray, false);
130  m_fileType = m_cFlagSec;
131  }
132  //secl
133  else if (name().my_functions.IsEndWith(m_cIdentificationSecOrSecl[1]))
134  {
135  isOpened = true;
136  File::SecFileRW::ReadLocked(name(), m_textArray, false);
137  m_fileType = m_cFlagSecl;
138  }
139  //sebt
140  else if (name().my_functions.IsEndWith(m_cIdentificationSebtOrSebtl[0]))
141  {
142  isOpened = true;
143  File::SebtFileRW::ReadNotLocked(name(), m_textNotArray, false);
144  m_fileType = m_cFlagSebt;
145  }
146  //sebtl
147  else if (name().my_functions.IsEndWith(m_cIdentificationSebtOrSebtl[1]))
148  {
149  isOpened = true;
150  File::SebtFileRW::ReadLocked(name(), m_textNotArray, false);
151  m_fileType = m_cFlagSebtl;
152  }
153  //csv
154  else if (name().my_functions.IsEndWith(m_cIdentificationCsv))
155  {
156  isOpened = true;
157  File::CsvFileRW::Read(name(), m_textArray, false);
158  m_fileType = m_cFlagCsv;
159  }
160  //others
161  else
162  {
163  for (int i = 0; i < 9; i++)
164  if (name().my_functions.IsEndWith(m_cIdentificationOthers[i]))

166  isOpened = true;

168  m_fileType = m_cFlagOthers;
169  }
170  }

172  //BaseAsset

174  //拡張子エラー
