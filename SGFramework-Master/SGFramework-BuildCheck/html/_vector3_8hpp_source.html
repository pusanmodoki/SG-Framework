439  //return: 自身とVector(0,0,0)の距離
440  inline constexpr float Vector3::Magnitude() const noexcept
441  {
442  float f = x * x + y * y + z * z;
443  int i = 0x5f3759df - (*reinterpret_cast<int*>(&f) >> 1);
444  float bitF = *reinterpret_cast<float*>(&i);
445 
446  bitF *= (1.5f - 0.5f * f * bitF * bitF);
447  return f * bitF * (1.5f - 0.5f * f * bitF * bitF);
448  }
